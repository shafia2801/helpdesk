# --- build stage: build the jar with Maven ---
FROM maven:3.9.4-eclipse-temurin-17 AS build
WORKDIR /workspace

# copy only what is needed to download dependencies first (speeds up rebuilds)
COPY pom.xml ./
COPY .mvn .mvn
COPY mvnw mvnw
RUN chmod +x mvnw || true
RUN mvn -B -f pom.xml dependency:go-offline

# copy source and build
COPY src ./src
RUN mvn -B clean package -DskipTests

# --- runtime stage: small JRE image to run the jar ---
FROM eclipse-temurin:17-jre-alpine
WORKDIR /app

# copy artifact (wildcard handles snapshot version)
COPY --from=build /workspace/target/*.jar app.jar

# Let Spring Boot pick the PORT env variable provided by Render
ENV JAVA_OPTS=""

EXPOSE 8080
ENTRYPOINT ["sh", "-c", "java $JAVA_OPTS -jar app.jar"]
